51 def distanceMisures(tmpSeqX, tmpConcX, tmpSeqY, tmpConcY, tmpIDs):
52  '''Function to compute the angle between two multidimensional vectors'''
53  strtoW = [0,0,0]
54  if tmpIDs != 0:
55  # COS BETWEEN VECTORS
56  tmpAllList = list(set(tmpSeqX) | set(tmpSeqY))
57  tmpAllList.sort()
58  speciesConcX = []
59  speciesConcY = []
60  for key in tmpAllList:
61  try:
62  posX = tmpSeqX.index(key)
63  speciesConcX.append(tmpConcX[posX])
64  except:
65  speciesConcX.append(0)
66  try:
67  posY = tmpSeqY.index(key)
68  speciesConcY.append(tmpConcY[posY])
69  except:
70  speciesConcY.append(0)
71 
72  vecX = np.array(speciesConcX)
73  vecY = np.array(speciesConcY)
74  tmpCos = float(np.dot(vecX,vecY) / (np.linalg.norm(vecX) * np.linalg.norm(vecY)))
75  if tmpCos >= 1.0: tmpCos = 1.0
76  strtoW[0] = np.arccos(tmpCos) * 360.00 / 6.28
77 
78  # HAMMING DISTANCE and EUCLIDEAN DISTANCE
79  tmpHD = 0
80  tmpEU = 0
81  for pos, x in enumerate(speciesConcX):
82  if ((x > 0.0) & (speciesConcY[pos] == 0.0)) | ((x == 0.0) & (speciesConcY[pos] > 0.0)):
83  tmpHD += 1
84  tmpEU += pow(x - speciesConcY[pos],2)
85 
86  strtoW[1] = tmpHD
87  strtoW[2] = pow(tmpEU,0.5)
88 
89  else:
90  strtoW[0] = 1
91  strtoW[1] = 0
92  strtoW[2] = 0
93 
94  return strtoW
