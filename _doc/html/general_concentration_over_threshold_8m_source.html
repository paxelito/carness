545  alreadyAdded_leaves = 0;
546  noInAcs = 1;
547  if realSccs > 0 % If there are ACS
548  for c = 1 : length(numScc) % for each ACS (if present)
549  if numScc(c) > 1 % IF the ACS contains more than one species
550  inSCCFlag = 0;
551  if sum(find(scc==c)==IDsOverThreshold(idot)) % If the species belong to the ACS
552  inSCCFlag = 1;
553  if alreadyAdded_ACS == 0 % Add concentration to conc of ACSs
554  conc_inSCC = conc_inSCC + concVec(idot);
555  alreadyAdded_ACS = 1;
556  end
557  end
558  % Reactions to distribuite in the
559  % different nature of the reactions
560  weightToDistribute = graph(and((graph(:,1) == incomingNodes(innode)),(graph(:,2) == IDsOverThreshold(idot))),6);
561  if sum(find(scc==c)==incomingNodes(innode)) % If the incoming node is in the ACS
562  noInAcs = 0;
563  if inSCCFlag == 1 % If the node is in an ACS, then it has been produced within an ACS
564  prod_inSCC = prod_inSCC + 1;
565  tmpProd_chain = tmpProd_chain - 1;
566  disp(fprintf('\t Within Acs %d -> %d #%d-[%d]%d %6.4f',incomingNodes(innode),IDsOverThreshold(idot),weightToDistribute,...
567  IDsOverThreshold(idot),IDsOverThreshold(idot),concVec(idot)));
568  prod_inSCC_weight = prod_inSCC_weight + weightToDistribute;
569  else % Otherwise it has been produced by an ACS, so it is a first layer leaf
570  if alreadyAdded_leaves == 0
571  conc_bySCC = conc_bySCC + concVec(idot);
572  alreadyAdded_leaves = 1;
573  end
574  prod_bySCC = prod_bySCC + 1;
575  tmpProd_chain = tmpProd_chain - 1;
576  disp(fprintf('\t From Acs %d -> %d #%d-[%d]%d %6.4f',incomingNodes(innode),IDsOverThreshold(idot),weightToDistribute,...
577  IDsOverThreshold(idot),IDsOverThreshold(idot),concVec(idot)));
578  prod_bySCC_weight = prod_bySCC_weight + weightToDistribute;
579  end
580  end
581  end
582  end
583  end
584  % If both the species over threshold and the
585  % incoming node are not belonging to an ACS
586  if noInAcs == 1
587  prod_chain_weight = prod_chain_weight + graph(and((graph(:,1) == incomingNodes(innode)),(graph(:,2) == IDsOverThreshold(idot))),6);
588  if alreadyAdded_chain == 0
589  conc_chain = conc_chain + concVec(idot);
590  alreadyAdded_chain = 1;
591  end
592  end
593  end
594  end
595  if self > 0
596  if sum(selfID==idot) > 0
597  wasteSpeciesFLAG = 1;
598  self_loop_weight = self_loop_weight + graph(and((graph(:,1) == IDsOverThreshold(idot)),(graph(:,2) == IDsOverThreshold(idot))),6);
599  conc_selfCat = conc_selfCat + concVec(idot);
600  autocatalysis = autocatalysis + 1;
601  end
602  end
603  % If the species concentration > 0 but the species is
604  % not produced by other species, so it is waste
605  if wasteSpeciesFLAG == 0
606  wasteSpecies = wasteSpecies + 1;
607  end
608 
609  % Compute the overlap between the different counter
610  prod_overlap = prod_overlap + min(min([prod_bySCC, prod_inSCC, tmpProd_chain]));
611  prod_chain = prod_chain + tmpProd_chain;
612  prod_overlap_weight = prod_overlap_weight + min(min([prod_inSCC_weight, prod_bySCC_weight, prod_chain_weight]));
613  end
614  end
615  disp(fprintf('\t\t<> Number of Structural Autocatalytic set of molecules: %d', realSccs));
616  disp(fprintf('\t\t<> -----------------------------------------------'));
617  disp(fprintf('\t\t<> Species over threshold produced by a CHAIN: %d', prod_chain));
618  if realSccs > 0
619  disp(fprintf('\t\t<> Species over threshold produced INTO an ACS: %d', prod_inSCC));
620  disp(fprintf('\t\t<> Species over threshold produced INTO an ACS (weigthed): %d', prod_inSCC_weight));
621  disp(fprintf('\t\t<> Species over threshold produced BY an ACS: %d', prod_bySCC));
622  disp(fprintf('\t\t<> Species over threshold produced BY an ACS (weigthed): %d', prod_bySCC_weight));
623  disp(fprintf('\t\t<> Species over threshold produced by an overlap: %d', prod_overlap));
624  disp(fprintf('\t\t<> Species over threshold produced by an overlap (weigthed): %d', prod_overlap_weight));
625  disp(fprintf('\t\t<> Species over threshold produced by itself: %d', autocatalysis));
626  disp(fprintf('\t\t<> Species over threshold produced by itself (weigthed): %d', self_loop_weight));
627  disp(fprintf('\t\t<> Concentration in ACSs: %6.4f', conc_inSCC));
628  disp(fprintf('\t\t<> Concentration in ACSs leaves: %6.4f', conc_bySCC));
629  disp(fprintf('\t\t<> Concentration in chains: %6.4f', conc_chain));
630  disp(fprintf('\t\t<> Concentration of autocatalyst: %6.4f', conc_selfCat));
631  disp(fprintf('\t\t<> Number of endo condensations: %6.4f', endo_condensation_counter));
632  disp(fprintf('\t\t<> Number of condensations: %6.4f', condensation_counter));
633  disp(fprintf('\t\t<> Number of endo cleavages: %6.4f', endo_cleavage_counter));
634  disp(fprintf('\t\t<> Number of cleavages: %6.4f', cleavage_counter));
635  end
636  disp(fprintf('\t\t<> -----------------------------------------------'));
637  end
638

641  rct = confparams.reactionProbability;
642  ecc = confparams.ECConcentration;
643  idOt = length(IDsOverThreshold);
644  fprintf(fidFINAL,'\n%s\t%6.4f\t%6.4f\t%6.4f\t%d\t%d\t%d\t%d\t%d\t%d\t%d\t%d\t%d\t%d\t%d\t%d\t%6.4f\t%6.4f\t%6.4f\t%6.4f\t%d\t%d\t%d\t%d\t%d',...
645  simDirs(IDF).name,rct,ecc,idOt,realSccs,prod_inSCC,prod_bySCC,prod_chain,prod_overlap,autocatalysis,prod_inSCC_weight,...
646  prod_bySCC_weight,prod_chain_weight,prod_overlap_weight,self_loop_weight,conc_inSCC,conc_bySCC,conc_chain,conc_selfCat,endo_condensation_counter,...
647  condensation_counter,endo_cleavage_counter,cleavage_counter,wasteSpecies);
648  fclose(fidFINAL);
649

651  clear gillTimeSeries;
652

654  IDsimFOLDER = IDsimFOLDER + 1;
655 
656  cd(params.tmpPath); % Come back to the original folder
657 
658  end
